//저장소 연동하기
1. cmd창에서 해당 프로젝트 파일로 경로 이동
2. git clone (주소)
3. .git 있는 폴더에서 활동

//cmd 창에서 깃허브 저장소에 올리기
0. git status                  //상태 확인
1. git add (파일명)            //커밋할 파일 결정
1. git add *                  //전부 올리기
2. git commit -m "(일지내용)"  //commit 커밋 : 로컷 저장소(내 컴퓨터)에 반영
3. git push                   //push 푸쉬 : 원격 저장소(깃허브 Repository)에 반영
